\subsection{Proof of Proposition~\ref{prop:ratio-bias}}
\label{app:proof-ratio-bias}
\begin{pf}
Write $y = H_x(S) + \varepsilon_x$ and $y_0 = H_0(S)(1+u)$ with $u\defeq\varepsilon_0/H_0(S)$, where $\varepsilon_x,\varepsilon_0$ are independent, mean zero, and $\mathrm{Var}(\varepsilon_0)=\sigma_0^2(S)$; in particular $\varepsilon_x$ and $u$ are independent. Then
\[
    \frac{y}{y_0} = \frac{H_x(S)+\varepsilon_x}{H_0(S)}\cdot(1+u)^{-1}.
\]
Since $(H_x(S)+\varepsilon_x)$ and $(1+u)^{-1}$ are independent (conditional on $S,x$), the expectation of the product factors exactly:
\[
    \expect{y/y_0\mid S,x} = \frac{1}{H_0(S)}\expect{H_x(S)+\varepsilon_x}\cdot\expect{(1+u)^{-1}} = r(S,x)\cdot\expect{(1+u)^{-1}},
\]
using $\expect{\varepsilon_x}=0$ and $r(S,x)=H_x(S)/H_0(S)$. It remains to bound $\expect{(1+u)^{-1}}$.

\paragraph{Well-posedness.} The map $t\mapsto(1+t)^{-1}$ has a simple, non-integrable pole at $t=-1$: if $u$ has a density that is positive and continuous at $-1$ (e.g.\ $u$ Gaussian, or sub-Gaussian in the usual unbounded-support sense), then $\expect{|(1+u)^{-1}|}=\infty$, since $\int_{-\delta}^{\delta}|t|^{-1}\,dt=\infty$ for every $\delta>0$ and the density is bounded away from $0$ on such an interval around $t=-1$ — light tails at $\pm\infty$ do not control behavior at this interior singularity. So $\expect{(1+u)^{-1}}$ is finite only if $u$ is bounded away from $-1$. We therefore assume $|u|\le\kappa(S)$ almost surely for a constant $\kappa(S)\in[0,1)$ — equivalently $|\varepsilon_0|\le\kappa(S)H_0(S)$ surely, i.e.\ $y_0\ge(1-\kappa(S))H_0(S)>0$ surely. This strengthens \cref{sec:data}'s assumption that $y_0>0$ with high probability to $y_0>0$ surely, with an explicit margin, and replaces the (insufficient) light-tail condition used informally in earlier drafts.

\paragraph{Exact remainder.} For any real $u\neq-1$,
\[
    \frac{1}{1+u} = 1-u+u^2-\frac{u^3}{1+u},
\]
an algebraic identity: multiplying both sides by $1+u$ reduces it to $1+u^3=(1+u)(1-u+u^2)$, the standard sum-of-cubes factorization. Unlike a truncated Taylor series, there is no discarded remainder at this stage — the identity is exact for every admissible $u$. Since $|u|\le\kappa(S)<1$ a.s.\ bounds every term on the right, all terms below are integrable and linearity of expectation applies:
\[
    \expect{(1+u)^{-1}} = 1-\expect{u}+\expect{u^2}-\expect{\frac{u^3}{1+u}} = 1+\frac{\sigma_0^2(S)}{H_0(S)^2}-\expect{\frac{u^3}{1+u}},
\]
using $\expect{u}=0$ and $\expect{u^2}=\mathrm{Var}(u)=\sigma_0^2(S)/H_0(S)^2$.

\paragraph{Bounding the remainder.} Since $|u|\le\kappa(S)$ a.s., $1+u\ge1-\kappa(S)>0$ a.s., so
\[
    \left|\frac{u^3}{1+u}\right| = \frac{|u|^3}{1+u} \;\le\; \frac{|u|^3}{1-\kappa(S)} \;\le\; \frac{\kappa(S)\,u^2}{1-\kappa(S)} \quad\text{a.s.},
\]
using $|u|^3=|u|\cdot u^2\le\kappa(S)u^2$. By $|\expect{X}|\le\expect{|X|}$, monotonicity of expectation under the a.s.\ bound, and linearity,
\[
    \left|\expect{\frac{u^3}{1+u}}\right| \;\le\; \frac{\kappa(S)}{1-\kappa(S)}\,\expect{u^2} \;=\; \frac{\kappa(S)}{1-\kappa(S)}\cdot\frac{\sigma_0^2(S)}{H_0(S)^2}.
\]
Define $\eta(S)$ by $\expect{(1+u)^{-1}}=:\big(1+\sigma_0^2(S)/H_0(S)^2\big)(1+\eta(S))$; dividing the remainder bound by $1+\sigma_0^2(S)/H_0(S)^2\ge1$ gives $|\eta(S)|\le\frac{\kappa(S)}{1-\kappa(S)}\cdot\frac{\sigma_0^2(S)}{H_0(S)^2}$. Multiplying by $r(S,x)$ and substituting into $\expect{y/y_0\mid S,x}=r(S,x)\cdot\expect{(1+u)^{-1}}$ from above gives the claim.
\end{pf}

\begin{rmk}
The bound's dependence on $\kappa(S)$ alone cannot be improved: if $\kappa(S)=c\in(0,1)$ is fixed while $\sigma_0^2(S)/H_0(S)^2\to0$ — e.g.\ $u=\pm c$ each with probability $p/2$ and $u=0$ with probability $1-p$, as $p\to0^+$ — then $\expect{u^2}=c^2p\to0$ while a direct computation gives $\eta(S)=\Theta(\expect{u^2})=\Theta(\sigma_0^2(S)/H_0(S)^2)$, matching the bound's order exactly. The sharper rate $\Oh\big((\sigma_0^2(S)/H_0(S)^2)^{3/2}\big)$ obtained by a naive (non-rigorous) Taylor-expansion heuristic requires the extra assumption $\kappa(S)=\Oh(\sigma_0(S)/H_0(S))$ — basal noise bounded by a constant multiple of its own standard deviation, which rules out the rare, fixed-size deviations of the example above. Under that extra assumption, $\frac{\kappa(S)}{1-\kappa(S)}\cdot\frac{\sigma_0^2(S)}{H_0(S)^2}=\Oh\big((\sigma_0^2(S)/H_0(S)^2)^{3/2}\big)$.
\end{rmk}
